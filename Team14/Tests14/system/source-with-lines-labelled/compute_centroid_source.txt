procedure main {
    flag = 0;                           (1)
    call computeCentroid;               (2)
    call printResults;                  (3)
}
procedure readPoint {
    read x;                             (4)
    read y;                             (5)
}
procedure printResults {
    print flag;                         (6)
    print cenX;                         (7)
    print cenY;                         (8)
    print normSq;                       (9)
}
procedure computeCentroid {
    count = 0;                          (10)
    cenX = 0;                           (11)
    cenY = 0;                           (12)
    call readPoint;                     (13)
    while ((x != 0) && (y != 0)) {      (14)
        count = count + 1;              (15)
        cenX = cenX + x;                (16)
        cenY = cenY + y;                (17)
        call readPoint;                 (18)
    }
    if (count == 0) then {              (19)
        flag = 1;                       (20)
    } else {
        cenX = cenX / count;            (21)
        cenY = cenY / count;            (22)
    }
    normSq = cenX * cenX + cenY * cenY; (23)
}
